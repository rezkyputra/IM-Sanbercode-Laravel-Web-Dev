1. Buat Database
CREATE DATABASE library78;

2. Buat Table
- Genre
CREATE TABLE genres( id integer PRIMARY KEY AUTO_INCREMENT, name varchar(255) NOT null );
- Books
CREATE table books( id integer AUTO_INCREMENT PRIMARY KEY, title varchar(255) NOT null, stock integer NOT null, genre_id integer NOT null, CONSTRAINT fk_genre FOREIGN KEY (genre_id) REFERENCES genres(id));
